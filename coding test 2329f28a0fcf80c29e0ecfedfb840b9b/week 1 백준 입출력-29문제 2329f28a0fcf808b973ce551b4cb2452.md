- 11720번 : 숫자의 합
    - 문자열로 입력 받아서 파싱
    
    ```bash
    #include <iostream>
    using namespace std;
    
    int main()
    {
        int num, sum = 0;
        string s;
        cin >> num >> s;  
    
        for (int i = 0; i < num; i++)
        {
            sum += s[i] - '0';     
        }
    
        cout << sum;
        return 0;
    }
    
    ```
